Module rtic_monotonics::nrf::rtc

Monotonic implementation for the nRF Real Time Clocks (RTC).

§Example

use rtic_monotonics::nrf::rtc::prelude::*;
nrf_rtc0_monotonic!(Mono);

fn init() {
    // Start the monotonic
    Mono::start(rtc);
}

async fn usage() {
    loop {
         // Use the monotonic
         let timestamp = Mono::now();
         Mono::delay(100.millis()).await;
    }
}
